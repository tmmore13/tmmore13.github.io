import matplotlib.pyplot as plt
import numpy as np

# --- Create the Data ---
x = np.linspace(0, 2 * np.pi, 200)
y_sin = np.sin(x)
y_cos = np.cos(x)

# --- Create a Figure with Two Side-by-Side Subplots ---
# This command creates a figure and a set of subplots.
# (1, 2) means 1 row, 2 columns.
# figsize sets the total size of the figure in inches.
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 5))

# --- Configure and Plot on the First (Left) Graph ---
ax1.plot(x, y_sin, color='blue')
ax1.set_title('Graph 1: Sine Function')
ax1.set_xlabel('x-axis')
ax1.set_ylabel('sin(x)')
ax1.grid(True)

# --- Configure and Plot on the Second (Right) Graph ---
ax2.plot(x, y_cos, color='red')
ax2.set_title('Graph 2: Cosine Function')
ax2.set_xlabel('x-axis')
ax2.set_ylabel('cos(x)')
ax2.grid(True)

# --- Final Adjustments and Display ---
# This automatically adjusts subplot params so that the subplot(s) fits in to the figure area.
plt.tight_layout()

# This command displays the entire figure with both subplots.
plt.show()
