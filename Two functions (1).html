import matplotlib.pyplot as plt
import numpy as np

# --- Data from the image ---
months = ['January', 'February', 'March', 'April']
sales = [22000, 25500, 31000, 28500]

# --- Create the Graph ---
plt.figure(figsize=(10, 6))

# 1. Plot the actual sales data as a BAR CHART (Style 1)
plt.bar(months, sales, label='Actual Sales', color='skyblue')

# 2. Calculate and plot a TREND LINE (Style 2)
# Create numeric x-values for calculation (0, 1, 2, 3)
x_values = np.arange(len(months))
# Calculate the slope and intercept of the line of best fit
slope, intercept = np.polyfit(x_values, sales, 1)
# Calculate the y-values for the trend line
trend_line = slope * x_values + intercept
# Plot the trend line
plt.plot(months, trend_line, label='Sales Trend', color='red', linestyle='--', marker='o')

# --- Add Final Touches ---
plt.title('Monthly Sales Performance with Trend')
plt.ylabel('Sales (USD)')
plt.legend()
plt.grid(axis='y', linestyle=':', alpha=0.7)

# Display the final plot
plt.show()

import matplotlib.pyplot as plt
import numpy as np

# --- Data from the image ---
months = ['January', 'February', 'March', 'April']
sales = [22000, 25500, 31000, 28500]

# --- Create a Figure with Two Side-by-Side Subplots ---
# This creates a figure with 1 row and 2 columns of plots
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 5))

# --- Plot 1: Actual Sales Bar Chart (on the left) ---
ax1.bar(months, sales, label='Actual Sales', color='skyblue')
ax1.set_title('Actual Sales Data')
ax1.set_ylabel('Sales (USD)')
ax1.grid(axis='y', linestyle=':', alpha=0.7)

# --- Plot 2: Trend Line (on the right) ---
# Calculate the trend line data first
x_values = np.arange(len(months))
slope, intercept = np.polyfit(x_values, sales, 1)
trend_line = slope * x_values + intercept
# Plot the trend line on the second axis
ax2.plot(months, trend_line, label='Sales Trend', color='red', linestyle='--', marker='o')
ax2.set_title('Sales Trend Line')
# Set the y-axis limits to match the bar chart for better comparison
ax2.set_ylim(bottom=0, top=max(sales) * 1.1)
ax2.grid(axis='y', linestyle=':', alpha=0.7)


# --- Final Touches ---
# Adjust the layout to prevent titles from overlapping
plt.tight_layout()
# Display the final figure containing both plots
plt.show()
