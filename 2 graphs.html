import matplotlib.pyplot as plt
import numpy as np

# --- Create the Data ---

# Create data for the smooth line plot (sine function)
x_line = np.linspace(0, 2 * np.pi, 200)
y_line = np.sin(x_line)

# Create data for the scatter plot (cosine function)
# We'll use fewer data points here so that the individual markers are clearly visible.
x_scatter = np.linspace(0, 2 * np.pi, 25)
y_scatter = np.cos(x_scatter)


# --- Create the Graph ---

plt.figure(figsize=(10, 6)) # Set the figure size

# 1. Plot the sine function as a solid blue LINE.
plt.plot(x_line, y_line, label='sin(x) [Line Style]', color='blue')

# 2. Plot the cosine function as a series of red DOTS (a scatter plot).
plt.scatter(x_scatter, y_scatter, label='cos(x) [Scatter Style]', color='red', marker='o')


# --- Add Final Touches ---

# Add title and labels
plt.title('Combining Plotting Styles')
plt.xlabel('x-axis')
plt.ylabel('y-axis')

# Add a grid and a legend to explain the styles
plt.grid(True)
plt.legend()

# Display the final plot
plt.show()
